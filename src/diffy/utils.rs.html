14/// Returns `true` if a byte must be quoted in a diff filename.
15///
16/// Matches git's behavior with all control characters
17/// (0x00-0x1f), DEL (0x7f), double quote, and backslash.
18pub(crate) fn byte_needs_quoting(b: u8) -> bool {
19    b < 0x20 || b == 0x7f || b == b'"' || b == b'\\'
20}

22/// Writes one byte in its escaped form to an [`io::Write`] sink.
23///
24/// Named escapes are used where git defines them (`\a`, `\b`,
25/// `\t`, `\n`, `\v`, `\f`, `\r`, `\\`, `\"`). Other bytes that
26/// require quoting are emitted as 3-digit octal (`\NNN`).
27/// Non-special bytes are written through unchanged.
28#[cfg(feature = "std")]
29pub(crate) fn write_escaped_byte<W: std::io::Write>(w: &mut W, b: u8) -> std::io::Result<()> {
30    match b {
31        b'\x07' => w.write_all(b"\\a"),
32        b'\x08' => w.write_all(b"\\b"),
33        b'\t' => w.write_all(b"\\t"),
34        b'\n' => w.write_all(b"\\n"),
35        b'\x0b' => w.write_all(b"\\v"),
36        b'\x0c' => w.write_all(b"\\f"),
37        b'\r' => w.write_all(b"\\r"),
38        b'"' => w.write_all(b"\\\""),
39        b'\\' => w.write_all(b"\\\\"),
40        b if b < 0x20 || b == 0x7f => {
41            let buf = [
42                b'\\',
43                b'0' + (b >> 6),
44                b'0' + ((b >> 3) & 7),
45                b'0' + (b & 7),
46            ];
47            w.write_all(&buf)
48        }
49        _ => w.write_all(&[b]),
50    }
51}

53/// Writes one byte in its escaped form to a [`fmt::Write`] sink.
54///
55/// Same logic as [`write_escaped_byte`] but for [`fmt::Write`].
56pub(crate) fn fmt_escaped_byte(f: &mut impl core::fmt::Write, b: u8) -> core::fmt::Result {
57    match b {
58        b'\x07' => f.write_str("\\a"),
59        b'\x08' => f.write_str("\\b"),
60        b'\t' => f.write_str("\\t"),
61        b'\n' => f.write_str("\\n"),
62        b'\x0b' => f.write_str("\\v"),
63        b'\x0c' => f.write_str("\\f"),
64        b'\r' => f.write_str("\\r"),
65        b'"' => f.write_str("\\\""),
66        b'\\' => f.write_str("\\\\"),
67        b if b < 0x20 || b == 0x7f => write!(f, "\\{:03o}", b),
68        _ => f.write_char(b as char),
69    }
70}

72/// Classifies lines, converting lines into unique `u64`s for quicker comparison
73pub struct Classifier<'a, T: ?Sized> {
74    next_id: u64,
75    unique_ids: HashMap<&'a T, u64>,
76}
77
78impl<'a, T: ?Sized + Eq + Hash> Classifier<'a, T> {
79    fn classify(&mut self, record: &'a T) -> u64 {
80        match self.unique_ids.entry(record) {
81            Entry::Occupied(o) => *o.get(),
82            Entry::Vacant(v) => {
83                let id = self.next_id;
84                self.next_id += 1;
85                *v.insert(id)
86            }
87        }
88    }
89}
90
91impl<'a, T: ?Sized + Text> Classifier<'a, T> {
92    pub fn classify_lines(&mut self, text: &'a T) -> (Vec<&'a T>, Vec<u64>) {
93        LineIter::new(text)
94            .map(|line| (line, self.classify(line)))
95            .unzip()
96    }
97}
98
99impl<T: Eq + Hash + ?Sized> Default for Classifier<'_, T> {
100    fn default() -> Self {
101        Self {
102            next_id: 0,
103            unique_ids: HashMap::default(),
104        }
105    }
106}

108/// Iterator over the lines of a string, including the `\n` character.
109pub struct LineIter<'a, T: ?Sized>(&'a T);
110
111impl<'a, T: ?Sized> LineIter<'a, T> {
112    pub fn new(text: &'a T) -> Self {
113        Self(text)
114    }
115}
116
117impl<'a, T: Text + ?Sized> Iterator for LineIter<'a, T> {
118    type Item = &'a T;
119
120    fn next(&mut self) -> Option<Self::Item> {
121        if self.0.is_empty() {
122            return None;
123        }
124
125        let end = if let Some(idx) = self.0.find("\n") {
126            idx + 1
127        } else {
128            self.0.len()
129        };
130
131        let (line, remaining) = self.0.split_at(end);
132        self.0 = remaining;
133        Some(line)
134    }
135}

137/// A helper trait for processing text like `str` and `[u8]`
138/// Useful for abstracting over those types for parsing as well as breaking input into lines
139pub trait Text: Eq + Hash + ToOwned {
140    fn is_empty(&self) -> bool;
141    fn len(&self) -> usize;
142    fn starts_with(&self, prefix: &str) -> bool;
143    fn ends_with(&self, suffix: &str) -> bool;
144    fn strip_prefix(&self, prefix: &str) -> Option<&Self>;
145    fn strip_suffix(&self, suffix: &str) -> Option<&Self>;
146    fn split_at_exclusive(&self, needle: &str) -> Option<(&Self, &Self)>;
147    fn find(&self, needle: &str) -> Option<usize>;
148    fn split_at(&self, mid: usize) -> (&Self, &Self);
149    fn as_str(&self) -> Option<&str>;
150    fn as_bytes(&self) -> &[u8];
151    fn lines(&self) -> LineIter<'_, Self>;
152
153    /// Converts raw bytes into `Self::Owned`.
154    ///
155    /// Returns `None` if the bytes are not valid for this type
156    /// (e.g. non-UTF-8 bytes for `str`).
157    fn owned_from_bytes(bytes: Vec<u8>) -> Option<Self::Owned>;
158
159    fn parse<T: core::str::FromStr>(&self) -> Option<T> {
160        self.as_str().and_then(|s| s.parse().ok())
161    }
162}

273fn find_bytes(haystack: &[u8], needle: &[u8]) -> Option<usize> {
274    match needle.len() {
275        0 => Some(0),
276        1 => find_byte(haystack, needle[0]),
277        len if len > haystack.len() => None,
278        needle_len => {
279            let mut offset = 0;
280            let mut haystack = haystack;
281
282            while let Some(position) = find_byte(haystack, needle[0]) {
283                offset += position;
284
285                if let Some(haystack) = haystack.get(position..position + needle_len) {
286                    if haystack == needle {
287                        return Some(offset);
288                    }
289                } else {
290                    return None;
291                }
292
293                haystack = &haystack[position + 1..];
294                offset += 1;
295            }
296
297            None
298        }
299    }
300}

307/// Decodes escape sequences in a quoted filename.
308///
309/// See [`byte_needs_quoting`] for the set of characters that
310/// require quoting.
311pub(crate) fn escaped_filename<T: Text + ?Sized>(
312    filename: &T,
313) -> Result<Cow<'_, T>, ParsePatchError> {
314    if let Some(inner) = filename
315        .strip_prefix("\"")
316        .and_then(|s| s.strip_suffix("\""))
317    {
318        match decode_escaped(inner.as_bytes())? {
319            None => Ok(Cow::Borrowed(inner)),
320            Some(bytes) => T::owned_from_bytes(bytes)
321                .map(Cow::Owned)
322                .ok_or_else(|| ParsePatchErrorKind::InvalidUtf8Path.into()),
323        }
324    } else {
325        let bytes = filename.as_bytes();
326        if bytes.iter().any(|b| byte_needs_quoting(*b)) {
327            return Err(ParsePatchErrorKind::InvalidCharInUnquotedFilename.into());
328        }
329        Ok(Cow::Borrowed(filename))
330    }
331}

333fn decode_escaped(bytes: &[u8]) -> Result<Option<Vec<u8>>, ParsePatchError> {
334    let mut result = Vec::new();
335    let mut i = 0;
336    let mut last_copy = 0;
337    let mut needs_allocation = false;
338
339    while i < bytes.len() {
340        if bytes[i] == b'\\' {
341            needs_allocation = true;
342            result.extend_from_slice(&bytes[last_copy..i]);
343
344            i += 1;
345            if i >= bytes.len() {
346                return Err(ParsePatchErrorKind::ExpectedEscapedChar.into());
347            }
348
349            let decoded = match bytes[i] {
350                b'a' => b'\x07',
351                b'b' => b'\x08',
352                b'n' => b'\n',
353                b't' => b'\t',
354                b'v' => b'\x0b',
355                b'f' => b'\x0c',
356                b'r' => b'\r',
357                b'\"' => b'\"',
358                b'\\' => b'\\',
359                // 3-digit octal: \0xx through \3xx (values 0x00–0xFF)
360                c @ b'0'..=b'3' => {
361                    if i + 2 >= bytes.len() {
362                        return Err(ParsePatchErrorKind::InvalidEscapedChar.into());
363                    }
364                    let d1 = bytes[i + 1];
365                    let d2 = bytes[i + 2];
366                    if !(b'0'..=b'7').contains(&d1) || !(b'0'..=b'7').contains(&d2) {
367                        return Err(ParsePatchErrorKind::InvalidEscapedChar.into());
368                    }
369                    i += 2;
370                    (c - b'0') << 6 | (d1 - b'0') << 3 | (d2 - b'0')
371                }
372                _ => return Err(ParsePatchErrorKind::InvalidEscapedChar.into()),
373            };
374            result.push(decoded);
375            i += 1;
376            last_copy = i;
377        } else if byte_needs_quoting(bytes[i]) {
378            return Err(ParsePatchErrorKind::InvalidUnescapedChar.into());
379        } else {
380            i += 1;
381        }
382    }
383
384    if needs_allocation {
385        result.extend_from_slice(&bytes[last_copy..]);
386        Ok(Some(result))
387    } else {
388        Ok(None)
389    }
