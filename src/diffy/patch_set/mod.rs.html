23/// Options for parsing patch content.
24///
25/// Use [`ParseOptions::unidiff()`] or [`ParseOptions::gitdiff()`]
26/// to create options for the desired format.
27///
28/// ## Binary Files
29///
30/// When parsing git diffs, binary file changes are detected by:
31///
32/// * `Binary files a/path and b/path differ` (`git diff` without `--binary` flag)
33/// * `GIT binary patch` (from `git diff --binary`)
34///
35/// Note that this is not a documented Git behavior,
36/// so the implementation here is subject to change if Git changes.
37///
38/// ## Example
39///
40/// ```
41/// use diffy::patch_set::ParseOptions;
42/// use diffy::patch_set::PatchSet;
43///
44/// let s = "\
45/// --- original
46/// +++ modified
47/// @@ -1 +1 @@
48/// -old
49/// +new
50/// ";
51///
52/// let patches: Vec<_> = PatchSet::parse(s, ParseOptions::unidiff())
53///     .collect::<Result<_, _>>()
54///     .unwrap();
55/// assert_eq!(patches.len(), 1);
56/// ```
57#[derive(Debug, Clone)]
58pub struct ParseOptions {
59    pub(crate) format: Format,
60}
61
62#[derive(Debug, Clone, Copy)]
63pub(crate) enum Format {
64    /// Standard unified diff format.
65    UniDiff,
66    /// Git extended diff format.
67    GitDiff,
68}
69
70impl ParseOptions {
71    /// Parse as standard [unified diff] format.
72    ///
73    /// Supported:
74    ///
75    /// * `---`/`+++` file headers
76    /// * `@@ ... @@` hunks
77    /// * modify and rename files
78    /// * create files (`--- /dev/null`)
79    /// * delete files (`+++ /dev/null`)
80    /// * Skip preamble, headers, and email signature trailer
81    ///
82    /// [unified diff]: https://www.gnu.org/software/diffutils/manual/html_node/Unified-Format.html
83    pub fn unidiff() -> Self {
84        Self {
85            format: Format::UniDiff,
86        }
87    }
88
89    /// Parse as [git extended diff format][git-diff-format].
90    ///
91    /// Supports all features of [`unidiff()`](Self::unidiff) plus:
92    ///
93    /// * `diff --git` headers
94    /// * Extended headers (`new file mode`, `deleted file mode`, etc.)
95    /// * Rename/copy detection (`rename from`/`rename to`, `copy from`/`copy to`)
96    /// * Binary file detection
97    ///
98    /// [git-diff-format]: https://git-scm.com/docs/diff-format
99    pub fn gitdiff() -> Self {
100        Self {
101            format: Format::GitDiff,
102        }
103    }
104}

106/// File mode extracted from git extended headers.
107#[derive(Debug, Clone, Copy, PartialEq, Eq)]
108pub enum FileMode {
109    /// `100644` regular file
110    Regular,
111    /// `100755` executable file
112    Executable,
113    /// `120000` symlink
114    Symlink,
115    /// `160000` gitlink (submodule)
116    Gitlink,
117}
118
119impl core::str::FromStr for FileMode {
120    type Err = PatchSetParseError;
121
122    fn from_str(mode: &str) -> Result<Self, Self::Err> {
123        match mode {
124            "100644" => Ok(Self::Regular),
125            "100755" => Ok(Self::Executable),
126            "120000" => Ok(Self::Symlink),
127            "160000" => Ok(Self::Gitlink),
128            _ => Err(PatchSetParseErrorKind::InvalidFileMode(mode.to_owned()).into()),
129        }
130    }
131}

133/// The kind of patch content in a [`FilePatch`].
134#[derive(Clone, PartialEq, Eq)]
135pub enum PatchKind<'a, T: ToOwned + ?Sized> {
136    /// Text patch with hunks.
137    Text(Patch<'a, T>),
138    /// Binary patch (literal or delta encoded, or marker-only).
139    Binary(BinaryPatch<'a>),
140}
141
142impl<T: ?Sized, O> fmt::Debug for PatchKind<'_, T>
143where
144    T: ToOwned<Owned = O> + fmt::Debug,
145    O: core::borrow::Borrow<T> + fmt::Debug,
146{
147    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
148        match self {
149            PatchKind::Text(patch) => f.debug_tuple("Text").field(patch).finish(),
150            PatchKind::Binary(patch) => f.debug_tuple("Binary").field(patch).finish(),
151        }
152    }
153}

155impl<'a, T: ToOwned + ?Sized> PatchKind<'a, T> {
156    /// Returns the text patch, or `None` if this is a binary patch.
157    pub fn as_text(&self) -> Option<&Patch<'a, T>> {
158        match self {
159            PatchKind::Text(patch) => Some(patch),
160            PatchKind::Binary(_) => None,
161        }
162    }
163
164    /// Returns the binary patch, or `None` if this is a text patch.
165    pub fn as_binary(&self) -> Option<&BinaryPatch<'a>> {
166        match self {
167            PatchKind::Binary(patch) => Some(patch),
168            PatchKind::Text(_) => None,
169        }
170    }
171
172    /// Returns `true` if this is a binary diff.
173    pub fn is_binary(&self) -> bool {
174        matches!(self, PatchKind::Binary(_))
175    }
176}

178/// A single file's patch with operation metadata.
179///
180/// This combines a [`PatchKind`] with a [`FileOperation`]
181/// that indicates what kind of file operation this patch represents
182/// (create, delete, modify, or rename).
183#[derive(Clone, PartialEq, Eq)]
184pub struct FilePatch<'a, T: ToOwned + ?Sized> {
185    operation: FileOperation<'a, T>,
186    kind: PatchKind<'a, T>,
187    old_mode: Option<FileMode>,
188    new_mode: Option<FileMode>,
189}

271/// The operation to perform based on a patch.
272///
273/// This is determined by examining the `---` and `+++` header lines
274/// of a unified diff patch, and git extended headers when available.
275#[derive(PartialEq, Eq)]
276pub enum FileOperation<'a, T: ToOwned + ?Sized> {
277    /// Delete a file (`+++ /dev/null`).
278    Delete(Cow<'a, T>),
279    /// Create a new file (`--- /dev/null`).
280    Create(Cow<'a, T>),
281    /// Modify a file.
282    ///
283    /// * If `original == modified`, this is an in-place modification.
284    /// * If they differ, the caller decides how to handle, e.g., treat as rename or error.
285    ///
286    /// Usually, the caller needs to strip the prefix from the paths to determine.
287    Modify {
288        /// The original path before the modification.
289        original: Cow<'a, T>,
290        /// The resulting path after the modification.
291        modified: Cow<'a, T>,
292    },
293    /// Rename a file (move from `from` to `to`, delete `from`).
294    ///
295    /// Only produced when git extended headers explicitly indicate a rename.
296    Rename {
297        /// The source path before the rename.
298        from: Cow<'a, T>,
299        /// The destination path after the rename.
300        to: Cow<'a, T>,
301    },
302    /// Copy a file (copy from `from` to `to`, keep `from`).
303    ///
304    /// Only produced when git extended headers explicitly indicate a copy.
305    Copy {
306        /// The source path that is copied from.
307        from: Cow<'a, T>,
308        /// The destination path that is copied to.
309        to: Cow<'a, T>,
310    },
311}
312
313impl<T: ToOwned + ?Sized> Clone for FileOperation<'_, T> {
314    fn clone(&self) -> Self {
315        match self {
316            Self::Delete(p) => Self::Delete(p.clone()),
317            Self::Create(p) => Self::Create(p.clone()),
318            Self::Modify { original, modified } => Self::Modify {
319                original: original.clone(),
320                modified: modified.clone(),
321            },
322            Self::Rename { from, to } => Self::Rename {
323                from: from.clone(),
324                to: to.clone(),
325            },
326            Self::Copy { from, to } => Self::Copy {
327                from: from.clone(),
328                to: to.clone(),
329            },
330        }
331    }
332}

362impl<T: Text + ?Sized> FileOperation<'_, T> {
363    /// Strip the first `n` path components from the paths in this operation.
364    ///
365    /// This is similar to the `-p` option in GNU patch. For example,
366    /// `strip_prefix(1)` on a path `a/src/lib.rs` would return `src/lib.rs`.
367    pub fn strip_prefix(&self, n: usize) -> FileOperation<'_, T> {
368        fn strip<T: Text + ?Sized>(path: &T, n: usize) -> &T {
369            let mut remaining = path;
370            for _ in 0..n {
371                match remaining.split_at_exclusive("/") {
372                    Some((_first, rest)) => remaining = rest,
373                    None => return remaining,
374                }
375            }
376            remaining
377        }
378
379        match self {
380            FileOperation::Delete(path) => FileOperation::Delete(Cow::Borrowed(strip(path, n))),
381            FileOperation::Create(path) => FileOperation::Create(Cow::Borrowed(strip(path, n))),
382            FileOperation::Modify { original, modified } => FileOperation::Modify {
383                original: Cow::Borrowed(strip(original, n)),
384                modified: Cow::Borrowed(strip(modified, n)),
385            },
386            FileOperation::Rename { from, to } => FileOperation::Rename {
387                from: Cow::Borrowed(strip(from, n)),
388                to: Cow::Borrowed(strip(to, n)),
389            },
390            FileOperation::Copy { from, to } => FileOperation::Copy {
391                from: Cow::Borrowed(strip(from, n)),
392                to: Cow::Borrowed(strip(to, n)),
393            },
394        }
395    }
396
397    /// Returns `true` if this is a file creation operation.
398    pub fn is_create(&self) -> bool {
399        matches!(self, FileOperation::Create(_))
400    }
401
402    /// Returns `true` if this is a file deletion operation.
403    pub fn is_delete(&self) -> bool {
404        matches!(self, FileOperation::Delete(_))
405    }
406
407    /// Returns `true` if this is a file modification.
408    pub fn is_modify(&self) -> bool {
409        matches!(self, FileOperation::Modify { .. })
410    }
411
412    /// Returns `true` if this is a rename operation.
413    pub fn is_rename(&self) -> bool {
414        matches!(self, FileOperation::Rename { .. })
415    }
416
417    /// Returns `true` if this is a copy operation.
418    pub fn is_copy(&self) -> bool {
419        matches!(self, FileOperation::Copy { .. })
420    }
