11/// An error returned when [`apply`]ing a `Patch` fails
12///
13/// [`apply`]: fn.apply.html
14#[derive(Debug)]
15pub struct ApplyError(usize);
16
17impl fmt::Display for ApplyError {
18    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
19        write!(f, "error applying hunk #{}", self.0)
20    }
21}

23impl core::error::Error for ApplyError {}

25#[derive(Debug)]
26enum ImageLine<'a, T: ?Sized> {
27    Unpatched(&'a T),
28    Patched(&'a T),
29}
30
31impl<'a, T: ?Sized> ImageLine<'a, T> {
32    fn inner(&self) -> &'a T {
33        match self {
34            ImageLine::Unpatched(inner) | ImageLine::Patched(inner) => inner,
35        }
36    }
37
38    fn into_inner(self) -> &'a T {
39        self.inner()
40    }
41
42    fn is_patched(&self) -> bool {
43        match self {
44            ImageLine::Unpatched(_) => false,
45            ImageLine::Patched(_) => true,
46        }
47    }
48}
49
50impl<T: ?Sized> Copy for ImageLine<'_, T> {}
51
52impl<T: ?Sized> Clone for ImageLine<'_, T> {
53    fn clone(&self) -> Self {
54        *self
55    }
56}

58/// Apply a `Patch` to a base image
59///
60/// ```
61/// use diffy::Patch;
62/// use diffy::apply;
63///
64/// let s = "\
65/// --- a/ideals
66/// +++ b/ideals
67/// @@ -1,4 +1,6 @@
68///  First:
69///      Life before death,
70///      strength before weakness,
71///      journey before destination.
72/// +Second:
73/// +    I will protect those who cannot protect themselves.
74/// ";
75///
76/// let patch = Patch::from_str(s).unwrap();
77///
78/// let base_image = "\
79/// First:
80///     Life before death,
81///     strength before weakness,
82///     journey before destination.
83/// ";
84///
85/// let expected = "\
86/// First:
87///     Life before death,
88///     strength before weakness,
89///     journey before destination.
90/// Second:
91///     I will protect those who cannot protect themselves.
92/// ";
93///
94/// assert_eq!(apply(base_image, &patch).unwrap(), expected);
95/// ```
96pub fn apply(base_image: &str, patch: &Patch<'_, str>) -> Result<String, ApplyError> {
97    let mut image: Vec<_> = LineIter::new(base_image)
98        .map(ImageLine::Unpatched)
99        .collect();
100
101    for (i, hunk) in patch.hunks().iter().enumerate() {
102        apply_hunk(&mut image, hunk).map_err(|_| ApplyError(i + 1))?;
103    }
104
105    Ok(image.into_iter().map(ImageLine::into_inner).collect())
106}

108/// Apply a non-utf8 `Patch` to a base image
109///
110/// # Examples
111///
112/// ```
113/// use diffy::Patch;
114/// use diffy::apply_bytes;
115///
116/// let patch = Patch::from_bytes(
117///     b"\
118/// --- a/ideals
119/// +++ b/ideals
120/// @@ -1 +1,2 @@
121///  First Ideal
122/// +Second Ideal
123/// ",
124/// )
125/// .unwrap();
126///
127/// let applied = apply_bytes(b"First Ideal\n", &patch).unwrap();
128///
129/// assert_eq!(
130///     applied,
131///     b"\
132/// First Ideal
133/// Second Ideal
134/// ",
135/// );
136/// ```
137pub fn apply_bytes(base_image: &[u8], patch: &Patch<'_, [u8]>) -> Result<Vec<u8>, ApplyError> {
138    let mut image: Vec<_> = LineIter::new(base_image)
139        .map(ImageLine::Unpatched)
140        .collect();
141
142    for (i, hunk) in patch.hunks().iter().enumerate() {
143        apply_hunk(&mut image, hunk).map_err(|_| ApplyError(i + 1))?;
144    }
145
146    Ok(image
147        .into_iter()
148        .flat_map(ImageLine::into_inner)
149        .copied()
150        .collect())
151}

153fn apply_hunk<'a, T: PartialEq + ?Sized>(
154    image: &mut Vec<ImageLine<'a, T>>,
155    hunk: &Hunk<'a, T>,
156) -> Result<(), ()> {
157    // Find position
158    let pos = find_position(image, hunk).ok_or(())?;
159
160    // update image
161    image.splice(
162        pos..pos + pre_image_line_count(hunk.lines()),
163        post_image(hunk.lines()).map(ImageLine::Patched),
164    );
165
166    Ok(())
167}
168
169// Search in `image` for a palce to apply hunk.
170// This follows the general algorithm (minus fuzzy-matching context lines) described in GNU patch's
171// man page.
172//
173// It might be worth looking into other possible positions to apply the hunk to as described here:
174// https://neil.fraser.name/writing/patch/
175fn find_position<T: PartialEq + ?Sized>(
176    image: &[ImageLine<T>],
177    hunk: &Hunk<'_, T>,
178) -> Option<usize> {
179    // In order to avoid searching through positions which are out of bounds of the image,
180    // clamp the starting position based on the length of the image
181    let pos = cmp::min(hunk.new_range().start().saturating_sub(1), image.len());
182
183    // Create an iterator that starts with 'pos' and then interleaves
184    // moving pos backward/foward by one.
185    let backward = (0..pos).rev();
186    let forward = pos + 1..image.len();
187
188    iter::once(pos)
189        .chain(interleave(backward, forward))
190        .find(|&pos| match_fragment(image, hunk.lines(), pos))
191}
192
193fn pre_image_line_count<T: ?Sized>(lines: &[Line<'_, T>]) -> usize {
194    pre_image(lines).count()
195}
196
197fn post_image<'a, 'b, T: ?Sized>(lines: &'b [Line<'a, T>]) -> impl Iterator<Item = &'a T> + 'b {
198    lines.iter().filter_map(|line| match line {
199        Line::Context(l) | Line::Insert(l) => Some(*l),
200        Line::Delete(_) => None,
201    })
202}
203
204fn pre_image<'a, 'b, T: ?Sized>(lines: &'b [Line<'a, T>]) -> impl Iterator<Item = &'a T> + 'b {
205    lines.iter().filter_map(|line| match line {
206        Line::Context(l) | Line::Delete(l) => Some(*l),
207        Line::Insert(_) => None,
208    })
209}
210
211fn match_fragment<T: PartialEq + ?Sized>(
212    image: &[ImageLine<T>],
213    lines: &[Line<'_, T>],
214    pos: usize,
215) -> bool {
216    let len = pre_image_line_count(lines);
217
218    let image = if let Some(image) = image.get(pos..pos + len) {
219        image
220    } else {
221        return false;
222    };
223
224    // If any of these lines have already been patched then we can't match at this position
225    if image.iter().any(ImageLine::is_patched) {
226        return false;
227    }
228
229    pre_image(lines).eq(image.iter().map(ImageLine::inner))
230}

232#[derive(Debug)]
233struct Interleave<I, J> {
234    a: iter::Fuse<I>,
235    b: iter::Fuse<J>,
236    flag: bool,
237}
238
239fn interleave<I, J>(
240    i: I,
241    j: J,
242) -> Interleave<<I as IntoIterator>::IntoIter, <J as IntoIterator>::IntoIter>
243where
244    I: IntoIterator,
245    J: IntoIterator<Item = I::Item>,
246{
247    Interleave {
248        a: i.into_iter().fuse(),
249        b: j.into_iter().fuse(),
250        flag: false,
251    }
252}
253
254impl<I, J> Iterator for Interleave<I, J>
255where
256    I: Iterator,
257    J: Iterator<Item = I::Item>,
258{
259    type Item = I::Item;
260
261    fn next(&mut self) -> Option<I::Item> {
262        self.flag = !self.flag;
263        if self.flag {
264            match self.a.next() {
265                None => self.b.next(),
266                item => item,
267            }
268        } else {
269            match self.b.next() {
270                None => self.a.next(),
271                item => item,
272            }
273        }
274    }
275}
