21enum Diff<'a, T: ?Sized> {
22    Equal(&'a T),
23    Delete(&'a T),
24    Insert(&'a T),
25}
26
27impl<T: ?Sized> Copy for Diff<'_, T> {}
28
29impl<T: ?Sized> Clone for Diff<'_, T> {
30    fn clone(&self) -> Self {
31        *self
32    }
33}

35impl<'a, T> From<DiffRange<'a, 'a, T>> for Diff<'a, T>
36where
37    T: ?Sized + SliceLike,
38{
39    fn from(diff: DiffRange<'a, 'a, T>) -> Self {
40        match diff {
41            DiffRange::Equal(range, _) => Diff::Equal(range.as_slice()),
42            DiffRange::Delete(range) => Diff::Delete(range.as_slice()),
43            DiffRange::Insert(range) => Diff::Insert(range.as_slice()),
44        }
45    }
46}

48/// A collection of options for modifying the way a diff is performed
49///
50/// # Examples
51///
52/// ```
53/// use diffy::DiffOptions;
54///
55/// let mut options = DiffOptions::new();
56/// options
57///     .set_context_len(1)
58///     .set_original_filename("before.txt")
59///     .set_modified_filename("after.txt");
60///
61/// let patch = options.create_patch(
62///     "\
63/// alpha
64/// beta
65/// gamma
66/// ",
67///     "\
68/// alpha
69/// BETA
70/// gamma
71/// ",
72/// );
73///
74/// assert_eq!(
75///     patch.to_string(),
76///     "\
77/// --- before.txt
78/// +++ after.txt
79/// @@ -1,3 +1,3 @@
80///  alpha
81/// -beta
82/// +BETA
83///  gamma
84/// ",
85/// );
86/// ```
87#[derive(Debug)]
88pub struct DiffOptions {
89    compact: bool,
90    context_len: usize,
91    original_filename: Option<Cow<'static, str>>,
92    modified_filename: Option<Cow<'static, str>>,
93}

147    // TODO determine if this should be exposed in the public API
148    fn diff<'a>(&self, original: &'a str, modified: &'a str) -> Vec<Diff<'a, str>> {
149        let solution = myers::diff(original.as_bytes(), modified.as_bytes());
150
151        let mut solution = solution
152            .into_iter()
153            .map(|diff_range| diff_range.to_str(original, modified))
154            .collect();
155
156        if self.compact {
157            cleanup::compact(&mut solution);
158        }
159
160        solution.into_iter().map(Diff::from).collect()
161    }
162
163    /// Produce a Patch between two texts based on the configured options
164    pub fn create_patch<'a>(&self, original: &'a str, modified: &'a str) -> Patch<'a, str> {
165        let mut classifier = Classifier::default();
166        let (old_lines, old_ids) = classifier.classify_lines(original);
167        let (new_lines, new_ids) = classifier.classify_lines(modified);
168
169        let solution = self.diff_slice(&old_ids, &new_ids);
170
171        let hunks = to_hunks(&old_lines, &new_lines, &solution, self.context_len);
172        Patch::new(
173            self.original_filename.clone(),
174            self.modified_filename.clone(),
175            hunks,
176        )
177    }
178
179    /// Create a patch between two potentially non-utf8 texts
180    pub fn create_patch_bytes<'a>(
181        &self,
182        original: &'a [u8],
183        modified: &'a [u8],
184    ) -> Patch<'a, [u8]> {
185        let mut classifier = Classifier::default();
186        let (old_lines, old_ids) = classifier.classify_lines(original);
187        let (new_lines, new_ids) = classifier.classify_lines(modified);
188
189        let solution = self.diff_slice(&old_ids, &new_ids);
190
191        let hunks = to_hunks(&old_lines, &new_lines, &solution, self.context_len);
192
193        // helper function to convert a utf8 cow to a bytes cow
194        fn cow_str_to_bytes(cow: Cow<'static, str>) -> Cow<'static, [u8]> {
195            match cow {
196                Cow::Borrowed(b) => Cow::Borrowed(b.as_bytes()),
197                Cow::Owned(o) => Cow::Owned(o.into_bytes()),
198            }
199        }
200
201        Patch::new(
202            self.original_filename.clone().map(cow_str_to_bytes),
203            self.modified_filename.clone().map(cow_str_to_bytes),
204            hunks,
205        )
206    }
207
208    pub(crate) fn diff_slice<'a, T: PartialEq>(
209        &self,
210        old: &'a [T],
211        new: &'a [T],
212    ) -> Vec<DiffRange<'a, 'a, [T]>> {
213        let mut solution = myers::diff(old, new);
214
215        if self.compact {
216            cleanup::compact(&mut solution);
217        }
218
219        solution
220    }
221}

297fn to_hunks<'a, T: ?Sized>(
298    lines1: &[&'a T],
299    lines2: &[&'a T],
300    solution: &[DiffRange<[u64]>],
301    context_len: usize,
302) -> Vec<Hunk<'a, T>> {
303    let edit_script = build_edit_script(solution);
304
305    let mut hunks = Vec::new();
306
307    let mut idx = 0;
308    while let Some(mut script) = edit_script.get(idx) {
309        let start1 = script.old.start.saturating_sub(context_len);
310        let start2 = script.new.start.saturating_sub(context_len);
311
312        let (mut end1, mut end2) = calc_end(
313            context_len,
314            lines1.len(),
315            lines2.len(),
316            script.old.end,
317            script.new.end,
318        );
319
320        let mut lines = Vec::new();
321
322        // Pre-context
323        for line in lines2.get(start2..script.new.start).into_iter().flatten() {
324            lines.push(Line::Context(*line));
325        }
326
327        loop {
328            // Delete lines from text1
329            for line in lines1.get(script.old.clone()).into_iter().flatten() {
330                lines.push(Line::Delete(*line));
331            }
332
333            // Insert lines from text2
334            for line in lines2.get(script.new.clone()).into_iter().flatten() {
335                lines.push(Line::Insert(*line));
336            }
337
338            if let Some(s) = edit_script.get(idx + 1) {
339                // Check to see if we can merge the hunks
340                let start1_next =
341                    cmp::min(s.old.start, lines1.len() - 1).saturating_sub(context_len);
342                if start1_next < end1 {
343                    // Context lines between hunks
344                    for (_i1, i2) in (script.old.end..s.old.start).zip(script.new.end..s.new.start)
345                    {
346                        if let Some(line) = lines2.get(i2) {
347                            lines.push(Line::Context(*line));
348                        }
349                    }
350
351                    // Calc the new end
352                    let (e1, e2) = calc_end(
353                        context_len,
354                        lines1.len(),
355                        lines2.len(),
356                        s.old.end,
357                        s.new.end,
358                    );
359
360                    end1 = e1;
361                    end2 = e2;
362                    script = s;
363                    idx += 1;
364                    continue;
365                }
366            }
367
368            break;
369        }
370
371        // Post-context
372        for line in lines2.get(script.new.end..end2).into_iter().flatten() {
373            lines.push(Line::Context(*line));
374        }
375
376        let len1 = end1 - start1;
377        let old_range = HunkRange::new(if len1 > 0 { start1 + 1 } else { start1 }, len1);
378
379        let len2 = end2 - start2;
380        let new_range = HunkRange::new(if len2 > 0 { start2 + 1 } else { start2 }, len2);
381
382        hunks.push(Hunk::new(old_range, new_range, None, lines));
383        idx += 1;
384    }
385
386    hunks
387}

389fn calc_end(
390    context_len: usize,
391    text1_len: usize,
392    text2_len: usize,
393    script1_end: usize,
394    script2_end: usize,
395) -> (usize, usize) {
396    let post_context_len = cmp::min(
397        context_len,
398        cmp::min(
399            text1_len.saturating_sub(script1_end),
400            text2_len.saturating_sub(script2_end),
401        ),
402    );
403
404    let end1 = script1_end + post_context_len;
405    let end2 = script2_end + post_context_len;
406
407    (end1, end2)
408}

422fn build_edit_script<T>(solution: &[DiffRange<[T]>]) -> Vec<EditRange> {
423    let mut idx_a = 0;
424    let mut idx_b = 0;
425
426    let mut edit_script: Vec<EditRange> = Vec::new();
427    let mut script = None;
428
429    for diff in solution {
430        match diff {
431            DiffRange::Equal(range1, range2) => {
432                idx_a += range1.len();
433                idx_b += range2.len();
434                if let Some(script) = script.take() {
435                    edit_script.push(script);
436                }
437            }
438            DiffRange::Delete(range) => {
439                match &mut script {
440                    Some(s) => s.old.end += range.len(),
441                    None => {
442                        script = Some(EditRange::new(idx_a..idx_a + range.len(), idx_b..idx_b));
443                    }
444                }
445                idx_a += range.len();
446            }
447            DiffRange::Insert(range) => {
448                match &mut script {
449                    Some(s) => s.new.end += range.len(),
450                    None => {
451                        script = Some(EditRange::new(idx_a..idx_a, idx_b..idx_b + range.len()));
452                    }
453                }
454                idx_b += range.len();
455            }
456        }
457    }
458
459    if let Some(script) = script.take() {
460        edit_script.push(script);
461    }
462
463    edit_script
464}
