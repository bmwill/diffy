16type Result<T, E = ParsePatchError> = core::result::Result<T, E>;

18/// Options that control parsing behavior.
19///
20/// Defaults match the [`parse`]/[`parse_bytes`] behavior.
21#[derive(Clone, Copy)]
22pub(crate) struct ParseOpts {
23    skip_preamble: bool,
24    reject_orphaned_hunks: bool,
25}
26
27impl Default for ParseOpts {
28    fn default() -> Self {
29        Self {
30            skip_preamble: true,
31            reject_orphaned_hunks: false,
32        }
33    }
34}
35
36impl ParseOpts {
37    /// Don't skip preamble lines before `---`/`+++`/`@@`.
38    ///
39    /// Useful when the caller has already positioned the input
40    /// at the start of the patch content.
41    pub(crate) fn no_skip_preamble(mut self) -> Self {
42        self.skip_preamble = false;
43        self
44    }
45
46    /// Reject orphaned `@@ ` hunk headers after parsed hunks,
47    /// matching `git apply` behavior.
48    pub(crate) fn reject_orphaned_hunks(mut self) -> Self {
49        self.reject_orphaned_hunks = true;
50        self
51    }
52}

54struct Parser<'a, T: Text + ?Sized> {
55    lines: core::iter::Peekable<LineIter<'a, T>>,
56    offset: usize,
57}
58
59impl<'a, T: Text + ?Sized> Parser<'a, T> {
60    fn new(input: &'a T) -> Self {
61        Self {
62            lines: LineIter::new(input).peekable(),
63            offset: 0,
64        }
65    }
66
67    fn peek(&mut self) -> Option<&&'a T> {
68        self.lines.peek()
69    }
70
71    fn offset(&self) -> usize {
72        self.offset
73    }
74
75    fn next(&mut self) -> Result<&'a T> {
76        let line = self
77            .lines
78            .next()
79            .ok_or_else(|| self.error(ParsePatchErrorKind::UnexpectedEof))?;
80        self.offset += line.len();
81        Ok(line)
82    }
83
84    /// Creates an error with the current offset as span.
85    fn error(&self, kind: ParsePatchErrorKind) -> ParsePatchError {
86        ParsePatchError::new(kind, self.offset..self.offset)
87    }
88
89    /// Creates an error with a specific offset as span.
90    fn error_at(&self, kind: ParsePatchErrorKind, offset: usize) -> ParsePatchError {
91        ParsePatchError::new(kind, offset..offset)
92    }
93}

95pub fn parse(input: &str) -> Result<Patch<'_, str>> {
96    let (result, _consumed) = parse_one(input, ParseOpts::default().reject_orphaned_hunks());
97    result
98}

100pub fn parse_bytes(input: &[u8]) -> Result<Patch<'_, [u8]>> {
101    let (result, _consumed) = parse_one(input, ParseOpts::default().reject_orphaned_hunks());
102    result

105/// Parses one patch from input.
106///
107/// Always returns consumed bytes alongside the result
108/// so callers can advance past the parsed or partially parsed content.
109pub(crate) fn parse_one<T: Text + ?Sized>(
110    input: &T,
111    opts: ParseOpts,
112) -> (Result<Patch<'_, T>>, usize) {
113    let mut parser = Parser::new(input);
114
115    let header = match patch_header(&mut parser, &opts) {
116        Ok(h) => h,
117        Err(e) => return (Err(e), parser.offset()),
118    };
119    let hunks = match hunks(&mut parser) {
120        Ok(h) => h,
121        Err(e) => return (Err(e), parser.offset()),
122    };
123    if opts.reject_orphaned_hunks {
124        if let Err(e) = reject_orphaned_hunk_headers(&mut parser) {
125            return (Err(e), parser.offset());
126        }
127    }
128
129    (Ok(Patch::new(header.0, header.1, hunks)), parser.offset())
130}

132#[expect(clippy::type_complexity)]
133fn patch_header<'a, T: Text + ?Sized>(
134    parser: &mut Parser<'a, T>,
135    opts: &ParseOpts,
136) -> Result<(Option<Cow<'a, T>>, Option<Cow<'a, T>>)> {
137    if opts.skip_preamble {
138        skip_header_preamble(parser)?;
139    }
140
141    let mut filename1 = None;
142    let mut filename2 = None;
143
144    while let Some(line) = parser.peek() {
145        if line.starts_with("--- ") {
146            if filename1.is_some() {
147                return Err(parser.error(ParsePatchErrorKind::MultipleOriginalHeaders));
148            }
149            filename1 = Some(parse_filename("--- ", parser.next()?)?);
150        } else if line.starts_with("+++ ") {
151            if filename2.is_some() {
152                return Err(parser.error(ParsePatchErrorKind::MultipleModifiedHeaders));
153            }
154            filename2 = Some(parse_filename("+++ ", parser.next()?)?);
155        } else {
156            break;
157        }
158    }
159
160    Ok((filename1, filename2))
161}

163// Skip to the first filename header ("--- " or "+++ ") or hunk line,
164// skipping any preamble lines like "diff --git", etc.
165fn skip_header_preamble<T: Text + ?Sized>(parser: &mut Parser<'_, T>) -> Result<()> {
166    while let Some(line) = parser.peek() {
167        if line.starts_with("--- ") | line.starts_with("+++ ") | line.starts_with("@@ ") {
168            break;
169        }
170        parser.next()?;
171    }
172
173    Ok(())
174}

176fn parse_filename<'a, T: Text + ?Sized>(prefix: &str, line: &'a T) -> Result<Cow<'a, T>> {
177    let line = line
178        .strip_prefix(prefix)
179        .ok_or(ParsePatchErrorKind::InvalidFilename)?;
180
181    let filename = if let Some((filename, _)) = line.split_at_exclusive("\t") {
182        filename
183    } else if let Some((filename, _)) = line.split_at_exclusive("\n") {
184        filename
185    } else {
186        return Err(ParsePatchErrorKind::FilenameUnterminated.into());
187    };
188
189    let filename = escaped_filename(filename)?;
190
191    Ok(filename)
192}

194fn verify_hunks_in_order<T: ?Sized>(hunks: &[Hunk<'_, T>]) -> bool {
195    for hunk in hunks.windows(2) {
196        if hunk[0].old_range.end() > hunk[1].old_range.start()
197            || hunk[0].new_range.end() > hunk[1].new_range.start()
198        {
199            return false;
200        }
201    }
202    true
203}

205/// Scans remaining lines for orphaned `@@ ` hunk headers.
206///
207/// In strict mode (git-apply behavior), trailing junk is allowed but
208/// an `@@ ` line hiding behind that junk indicates a lost hunk.
209fn reject_orphaned_hunk_headers<T: Text + ?Sized>(parser: &mut Parser<'_, T>) -> Result<()> {
210    while let Some(line) = parser.peek() {
211        if line.starts_with("@@ ") {
212            return Err(parser.error(ParsePatchErrorKind::OrphanedHunkHeader));
213        }
214        parser.next()?;
215    }
216    Ok(())
217}

219fn hunks<'a, T: Text + ?Sized>(parser: &mut Parser<'a, T>) -> Result<Vec<Hunk<'a, T>>> {
220    let mut hunks = Vec::new();
221
222    // Parse hunks while we see @@ headers.
223    //
224    // Following GNU patch behavior: stop at non-@@ content.
225    // Any trailing content (including hidden @@ headers) is silently ignored.
226    // This is more permissive than git apply, which errors on junk between hunks.
227    while parser.peek().is_some_and(|line| line.starts_with("@@ ")) {
228        hunks.push(hunk(parser)?);
229    }
230
231    // check and verify that the Hunks are in sorted order and don't overlap
232    if !verify_hunks_in_order(&hunks) {
233        return Err(parser.error(ParsePatchErrorKind::HunksOutOfOrder));
234    }
235
236    Ok(hunks)
237}

239fn hunk<'a, T: Text + ?Sized>(parser: &mut Parser<'a, T>) -> Result<Hunk<'a, T>> {
240    let hunk_start = parser.offset();
241    let header_line = parser.next()?;
242    let (range1, range2, function_context) =
243        hunk_header(header_line).map_err(|e| parser.error_at(e.kind, hunk_start))?;
244    let lines = hunk_lines(parser, range1.len, range2.len, hunk_start)?;
245
246    Ok(Hunk::new(range1, range2, function_context, lines))
247}

249fn hunk_header<T: Text + ?Sized>(input: &T) -> Result<(HunkRange, HunkRange, Option<&T>)> {
250    let input = input
251        .strip_prefix("@@ ")
252        .ok_or(ParsePatchErrorKind::InvalidHunkHeader)?;
253
254    let (ranges, function_context) = input
255        .split_at_exclusive(" @@")
256        .ok_or(ParsePatchErrorKind::HunkHeaderUnterminated)?;
257    let function_context = function_context.strip_prefix(" ");
258
259    let (range1, range2) = ranges
260        .split_at_exclusive(" ")
261        .ok_or(ParsePatchErrorKind::InvalidHunkHeader)?;
262    let range1 = range(
263        range1
264            .strip_prefix("-")
265            .ok_or(ParsePatchErrorKind::InvalidHunkHeader)?,
266    )?;
267    let range2 = range(
268        range2
269            .strip_prefix("+")
270            .ok_or(ParsePatchErrorKind::InvalidHunkHeader)?,
271    )?;
272    Ok((range1, range2, function_context))
273}

275fn range<T: Text + ?Sized>(s: &T) -> Result<HunkRange> {
276    let (start, len): (usize, usize) = if let Some((start, len)) = s.split_at_exclusive(",") {
277        (
278            start.parse().ok_or(ParsePatchErrorKind::InvalidRange)?,
279            len.parse().ok_or(ParsePatchErrorKind::InvalidRange)?,
280        )
281    } else {
282        (s.parse().ok_or(ParsePatchErrorKind::InvalidRange)?, 1)
283    };
284
285    // reject ranges that overflow
286    start
287        .checked_add(len)
288        .ok_or(ParsePatchErrorKind::InvalidRange)?;
289
290    Ok(HunkRange::new(start, len))
291}

293fn hunk_lines<'a, T: Text + ?Sized>(
294    parser: &mut Parser<'a, T>,
295    expected_old: usize,
296    expected_new: usize,
297    hunk_start: usize,
298) -> Result<Vec<Line<'a, T>>> {
299    let mut lines: Vec<Line<'a, T>> = Vec::new();
300    let mut no_newline_context = false;
301    let mut no_newline_delete = false;
302    let mut no_newline_insert = false;
303
304    // Track current line counts (old = context + delete, new = context + insert)
305    let mut old_count = 0;
306    let mut new_count = 0;
307
308    while let Some(line) = parser.peek() {
309        // Check if hunk is complete
310        let hunk_complete = old_count >= expected_old && new_count >= expected_new;
311
312        let line = if line.starts_with("@") {
313            break;
314        } else if no_newline_context {
315            // After `\ No newline at end of file` on a context line,
316            // only a new hunk header is valid. Any other line means
317            // the hunk should be complete, or it's an error.
318            if hunk_complete {
319                break;
320            }
321            return Err(parser.error(ParsePatchErrorKind::ExpectedEndOfHunk));
322        } else if let Some(line) = line.strip_prefix(" ") {
323            if hunk_complete {
324                break;
325            }
326            Line::Context(line)
327        } else if line.starts_with("\n") {
328            if hunk_complete {
329                break;
330            }
331            Line::Context(*line)
332        } else if let Some(line) = line.strip_prefix("-") {
333            if no_newline_delete {
334                return Err(parser.error(ParsePatchErrorKind::TooManyDeletedLines));
335            }
336            if hunk_complete {
337                break;
338            }
339            Line::Delete(line)
340        } else if let Some(line) = line.strip_prefix("+") {
341            if no_newline_insert {
342                return Err(parser.error(ParsePatchErrorKind::TooManyInsertedLines));
343            }
344            if hunk_complete {
345                break;
346            }
347            Line::Insert(line)
348        } else if line.starts_with(NO_NEWLINE_AT_EOF) {
349            // The `\ No newline at end of file` marker indicates
350            // the previous line doesn't end with a newline.
351            // It's not a content line itself.
352            // Therefore, we
353            //
354            // * strip the newline character of the previous line
355            // * don't increment line counts and continue to next directly
356            let last_line = lines
357                .pop()
358                .ok_or_else(|| parser.error(ParsePatchErrorKind::UnexpectedNoNewlineMarker))?;
359            let modified = match last_line {
360                Line::Context(line) => {
361                    no_newline_context = true;
362                    Line::Context(strip_newline(line)?)
363                }
364                Line::Delete(line) => {
365                    no_newline_delete = true;
366                    Line::Delete(strip_newline(line)?)
367                }
368                Line::Insert(line) => {
369                    no_newline_insert = true;
370                    Line::Insert(strip_newline(line)?)
371                }
372            };
373            lines.push(modified);
374            parser.next()?;
375            continue;
376        } else {
377            // Non-hunk line encountered
378            if hunk_complete {
379                // Hunk is complete, treat remaining content as garbage
380                break;
381            }
382            return Err(parser.error(ParsePatchErrorKind::UnexpectedHunkLine));
383        };
384
385        match &line {
386            Line::Context(_) => {
387                old_count += 1;
388                new_count += 1;
389            }
390            Line::Delete(_) => {
391                old_count += 1;
392            }
393            Line::Insert(_) => {
394                new_count += 1;
395            }
396        }
397
398        lines.push(line);
399        parser.next()?;
400    }
401
402    // Final check: ensure we got the expected number of lines
403    if old_count != expected_old || new_count != expected_new {
404        return Err(parser.error_at(ParsePatchErrorKind::HunkMismatch, hunk_start));
405    }
406
407    Ok(lines)
408}

410fn strip_newline<T: Text + ?Sized>(s: &T) -> Result<&T> {
411    if let Some(stripped) = s.strip_suffix("\n") {
412        Ok(stripped)
413    } else {
414        Err(ParsePatchErrorKind::MissingNewline.into())
415    }
