18/// Cap preallocation when the size comes from untrusted input.
19///
20/// This prevents instant OOM from a bogus header.
21/// The vec grows as needed if the actual output is larger.
22#[cfg(feature = "binary")]
23const MAX_PREALLOC: u64 = 64 * 1024; // 64 KiB
24
25/// The type of a binary patch block.
26#[derive(Debug, Clone, Copy, PartialEq, Eq)]
27pub enum BinaryBlockKind {
28    /// [Literal](https://diffx.org/spec/binary-diffs.html#git-literal-binary-diffs):
29    /// contains the full file content, zlib-compressed and Base85-encoded.
30    Literal,
31    /// [Delta](https://diffx.org/spec/binary-diffs.html#git-delta-binary-diffs):
32    /// contains delta instructions to transform one file into another.
33    Delta,
34}
35
36/// A single block in a binary patch.
37#[derive(Debug, Clone, PartialEq, Eq)]
38pub struct BinaryBlock<'a> {
39    /// The type of this block (literal content or delta instructions).
40    pub kind: BinaryBlockKind,
41    /// The encoded data.
42    pub data: BinaryData<'a>,
43}

45/// A parsed binary patch.
46///
47/// A binary patch contains encoded binary data that can be decoded
48/// to recover the original and modified file contents.
49///
50/// Git may use different encodings for each direction:
51///
52/// - `literal`: full file content
53/// - `delta`: instructions to transform one file into another
54#[derive(Debug, Clone, PartialEq, Eq)]
55pub enum BinaryPatch<'a> {
56    /// A full binary patch with forward and reverse data.
57    ///
58    /// The forward block transforms original -> modified.
59    /// The reverse block transforms modified -> original.
60    ///
61    /// Each block can independently be either `literal` or `delta`.
62    Full {
63        /// Forward transformation (original -> modified).
64        forward: BinaryBlock<'a>,
65        /// Reverse transformation (modified -> original).
66        reverse: BinaryBlock<'a>,
67    },
68    /// A Git binary diff marker.
69    ///
70    /// This represents the `Binary files a/path and b/path differ` case,
71    /// where git detected a binary change but didn't include the actual data.
72    ///
73    /// Calling [`apply()`](Self::apply) on this variant returns an error.
74    Marker,
75}
76
77impl<'a> BinaryPatch<'a> {
78    /// Applies a binary patch forward: original -> modified.
79    ///
80    /// - If the forward block is `Literal`: returns the decoded content directly.
81    /// - If the forward block is `Delta`: applies delta instructions to `original`.
82    ///
83    /// Unlike `git apply`, this doesn't validate the original content hash.
84    ///
85    /// # Examples
86    ///
87    /// ```
88    /// use diffy::patch_set::ParseOptions;
89    /// use diffy::patch_set::PatchSet;
90    ///
91    /// let input = b"\
92    /// diff --git a/blob b/blob
93    /// index e69de29..0000000 100644
94    /// GIT binary patch
95    /// literal 10
96    /// UcmV+l0QLU>0RjUA1qKHQ2>`DEE&u=k
97    ///
98    /// literal 0
99    /// KcmV+b0RR6000031
100    ///
101    /// ";
102    ///
103    /// let patch = PatchSet::parse_bytes(input, ParseOptions::gitdiff())
104    ///     .next()
105    ///     .unwrap()
106    ///     .unwrap();
107    /// let binary = patch.patch().as_binary().unwrap();
108    ///
109    /// assert_eq!(
110    ///     binary.apply(&[]).unwrap(),
111    ///     b"\0\x01\x02\x03\x04\x05\x06\x07\x08\t"
112    /// );
113    /// ```
114    #[cfg(feature = "binary")]
115    #[cfg_attr(docsrs, doc(cfg(feature = "binary")))]
116    pub fn apply(&self, original: &[u8]) -> Result<Vec<u8>, BinaryPatchParseError> {
117        match self {
118            BinaryPatch::Full { forward, .. } => Self::apply_block(forward, original),
119            BinaryPatch::Marker => Err(BinaryPatchParseErrorKind::NoBinaryData.into()),
120        }
121    }
122
123    /// Applies a binary patch in reverse: modified -> original.
124    ///
125    /// - If the reverse block is `Literal`: returns the decoded content directly.
126    /// - If the reverse block is `Delta`: applies delta instructions to `modified`.
127    ///
128    /// Unlike `git apply`, this doesn't validate the modified content hash.
129    ///
130    /// # Examples
131    ///
132    /// ```
133    /// use diffy::patch_set::ParseOptions;
134    /// use diffy::patch_set::PatchSet;
135    ///
136    /// let input = b"\
137    /// diff --git a/blob b/blob
138    /// index e69de29..0000000 100644
139    /// GIT binary patch
140    /// literal 10
141    /// UcmV+l0QLU>0RjUA1qKHQ2>`DEE&u=k
142    ///
143    /// literal 0
144    /// KcmV+b0RR6000031
145    ///
146    /// ";
147    ///
148    /// let patch = PatchSet::parse_bytes(input, ParseOptions::gitdiff())
149    ///     .next()
150    ///     .unwrap()
151    ///     .unwrap();
152    /// let binary = patch.patch().as_binary().unwrap();
153    ///
154    /// assert_eq!(binary.apply_reverse(&[]).unwrap(), b"");
155    /// ```
156    #[cfg(feature = "binary")]
157    #[cfg_attr(docsrs, doc(cfg(feature = "binary")))]
158    pub fn apply_reverse(&self, modified: &[u8]) -> Result<Vec<u8>, BinaryPatchParseError> {
159        match self {
160            BinaryPatch::Full { reverse, .. } => Self::apply_block(reverse, modified),
161            BinaryPatch::Marker => Err(BinaryPatchParseErrorKind::NoBinaryData.into()),
162        }
163    }
164
165    /// Applies a single block (either literal or delta).
166    #[cfg(feature = "binary")]
167    fn apply_block(block: &BinaryBlock<'_>, base: &[u8]) -> Result<Vec<u8>, BinaryPatchParseError> {
168        match block.kind {
169            BinaryBlockKind::Literal => Self::decode_data(&block.data),
170            BinaryBlockKind::Delta => {
171                let delta_instructions = Self::decode_data(&block.data)?;
172                delta::apply(base, &delta_instructions).map_err(BinaryPatchParseError::from)
173            }
174        }
175    }
176
177    /// See [Decoding Logic](https://diffx.org/spec/binary-diffs.html#decoding-logic)
178    #[cfg(feature = "binary")]
179    fn decode_data(binary_data: &BinaryData<'_>) -> Result<Vec<u8>, BinaryPatchParseError> {
180        use alloc::vec;
181        use zlib_rs::Inflate;
182        use zlib_rs::InflateFlush;
183        use zlib_rs::Status;
184
185        let compressed = decode_base85_lines(binary_data.data)?;
186
187        // Bound the initial allocation so a bogus header can't request
188        // gigabytes upfront. The output grows as inflation produces data.
189        let initial_len = binary_data.size.clamp(1, MAX_PREALLOC) as usize;
190        let mut output: Vec<u8> = vec![0; initial_len];
191
192        let mut inflate = Inflate::new(true, 15);
193        let mut in_pos = 0usize;
194        let mut out_pos = 0usize;
195
196        loop {
197            let prev_in = inflate.total_in();
198            let prev_out = inflate.total_out();
199
200            let status = inflate
201                .decompress(
202                    &compressed[in_pos..],
203                    &mut output[out_pos..],
204                    InflateFlush::Finish,
205                )
206                .map_err(|e| BinaryPatchParseErrorKind::DecompressionFailed(e.as_str()))?;
207
208            in_pos += (inflate.total_in() - prev_in) as usize;
209            out_pos += (inflate.total_out() - prev_out) as usize;
210
211            match status {
212                Status::StreamEnd => {
213                    output.truncate(out_pos);
214                    break;
215                }
216                Status::Ok | Status::BufError => {
217                    if out_pos == output.len() {
218                        // Output buffer is full: grow and keep inflating.
219                        let new_len = output.len().saturating_mul(2);
220                        output.resize(new_len, 0);
221                    }
222                }
223            }
224        }
225
226        if output.len() as u64 != binary_data.size {
227            return Err(BinaryPatchParseErrorKind::DecompressedSizeMismatch {
228                expected: binary_data.size,
229                actual: output.len() as u64,
230            }
231            .into());
232        }
233
234        Ok(output)
235    }
236}

238/// Represents a single binary payload in a Git binary diff.
239///
240/// For example, the following patch block
241///
242/// * is parsed as `BinaryData { size: 10, data: b"UcmV+l0QLU>0RjUA1qKHQ2>\`DEE&u=k" }`
243/// * The line starts with a length indicator (`U` = 21 decoded bytes)
244///
245///
246/// ```text
247/// literal 10
248/// UcmV+l0QLU>0RjUA1qKHQ2>`DEE&u=k
249/// ```
250#[derive(Debug, Clone, PartialEq, Eq)]
251pub struct BinaryData<'a> {
252    /// Uncompressed size in bytes.
253    pub size: u64,
254    /// Raw Base85 lines with length indicators.
255    pub data: &'a [u8],
256}

258/// Error type for binary patch operations.
259#[derive(Debug, Clone, PartialEq, Eq)]
260pub struct BinaryPatchParseError {
261    pub(crate) kind: BinaryPatchParseErrorKind,
262    span: Option<Range<usize>>,
263}
264
265impl BinaryPatchParseError {
266    /// Creates a new error with the given kind and span.
267    pub(crate) fn new(kind: BinaryPatchParseErrorKind, span: Range<usize>) -> Self {
268        Self {
269            kind,
270            span: Some(span),
271        }
272    }
273
274    /// Returns the byte range in the input where the error occurred.
275    #[expect(unused)]
276    pub(crate) fn span(&self) -> Option<Range<usize>> {
277        self.span.clone()
278    }
279}

281impl fmt::Display for BinaryPatchParseError {
282    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
283        if let Some(span) = &self.span {
284            write!(
285                f,
286                "error parsing binary patch at byte {}: {}",
287                span.start, self.kind
288            )
289        } else {
290            write!(f, "error parsing binary patch: {}", self.kind)
291        }
292    }
293}

317/// The kind of error that occurred when parsing a binary patch.
318#[derive(Debug, Clone, PartialEq, Eq)]
319#[non_exhaustive]
320pub(crate) enum BinaryPatchParseErrorKind {
321    /// Missing or invalid "GIT binary patch" header.
322    InvalidHeader,
323
324    /// First binary block (forward) not found.
325    MissingForwardBlock,
326
327    /// Second binary block (reverse) not found.
328    MissingReverseBlock,
329
330    /// No binary data available (marker-only patch).
331    #[cfg_attr(not(feature = "binary"), expect(dead_code))]
332    NoBinaryData,
333
334    /// Invalid line length indicator in Base85 data.
335    #[cfg_attr(not(feature = "binary"), expect(dead_code))]
336    InvalidLineLengthIndicator,
337
338    /// Base85 decoding failed.
339    #[cfg(feature = "binary")]
340    Base85(base85::Base85Error),
341
342    /// Delta application failed.
343    #[cfg(feature = "binary")]
344    Delta(delta::DeltaError),
345
346    /// Zlib decompression failed.
347    #[cfg(feature = "binary")]
348    DecompressionFailed(&'static str),
349
350    /// Decompressed size doesn't match declared size.
351    #[cfg(feature = "binary")]
352    DecompressedSizeMismatch { expected: u64, actual: u64 },
353}
354
355impl fmt::Display for BinaryPatchParseErrorKind {
356    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
357        match self {
358            Self::InvalidHeader => write!(f, "invalid binary patch header"),
359            Self::MissingForwardBlock => write!(f, "first binary block not found"),
360            Self::MissingReverseBlock => write!(f, "second binary block not found"),
361            Self::NoBinaryData => write!(f, "no binary data available"),
362            Self::InvalidLineLengthIndicator => write!(f, "invalid line length indicator"),
363            #[cfg(feature = "binary")]
364            Self::Base85(e) => write!(f, "{e}"),
365            #[cfg(feature = "binary")]
366            Self::Delta(e) => write!(f, "{e}"),
367            #[cfg(feature = "binary")]
368            Self::DecompressionFailed(msg) => write!(f, "decompression failed: {msg}"),
369            #[cfg(feature = "binary")]
370            Self::DecompressedSizeMismatch { expected, actual } => {
371                write!(
372                    f,
373                    "decompressed size mismatch: expected {expected}, got {actual}"
374                )
375            }
376        }
377    }
378}

380/// Simple streaming parser for binary patches.
381struct BinaryParser<'a> {
382    input: &'a [u8],
383    offset: usize,
384}
385
386impl<'a> BinaryParser<'a> {
387    fn new(input: &'a [u8]) -> Self {
388        Self { input, offset: 0 }
389    }
390
391    /// Creates an error with the current offset as span.
392    fn error(&self, kind: BinaryPatchParseErrorKind) -> BinaryPatchParseError {
393        BinaryPatchParseError::new(kind, self.offset..self.offset)
394    }
395
396    fn next_line(&mut self) -> Option<&'a [u8]> {
397        let rest = &self.input[self.offset..];
398        if rest.is_empty() {
399            return None;
400        }
401        let (line, skip) = match rest.iter().position(|&b| b == b'\n') {
402            Some(pos) => (&rest[..pos], pos + 1),
403            None => (rest, rest.len()),
404        };
405        self.offset += skip;
406        Some(line.strip_suffix(b"\r").unwrap_or(line))
407    }
408}

410/// Parses binary patch content after git extended headers.
411///
412/// Expects input starting with "GIT binary patch" line.
413/// Returns the parsed patch and the number of bytes consumed.
414///
415/// Format:
416///
417/// ```text
418/// GIT binary patch
419/// <literal|delta> <modified_size>
420/// <base85_lines>
421///
422/// <literal|delta> <original_size>
423/// <base85_lines>
424/// ```
425pub(crate) fn parse_binary_patch(
426    input: &[u8],
427) -> Result<(BinaryPatch<'_>, usize), BinaryPatchParseError> {
428    let mut parser = BinaryParser::new(input);
429
430    // Expect "GIT binary patch" marker
431    if parser.next_line() != Some(b"GIT binary patch".as_slice()) {
432        return Err(parser.error(BinaryPatchParseErrorKind::InvalidHeader));
433    }
434
435    // Parse first block (forward: original -> modified)
436    let Some(forward) = parse_binary_block(&mut parser) else {
437        return Err(parser.error(BinaryPatchParseErrorKind::MissingForwardBlock));
438    };
439
440    // Parse second block (reverse: modified -> original)
441    let Some(reverse) = parse_binary_block(&mut parser) else {
442        return Err(parser.error(BinaryPatchParseErrorKind::MissingReverseBlock));
443    };
444
445    Ok((BinaryPatch::Full { forward, reverse }, parser.offset))
446}

448/// Parses a single binary block.
449///
450/// Returns a `BinaryBlock` with kind (literal/delta) and data.
451fn parse_binary_block<'a>(parser: &mut BinaryParser<'a>) -> Option<BinaryBlock<'a>> {
452    // Parse "literal 10" or "delta 18"
453    let format_line = parser.next_line()?;
454    let space = format_line.iter().position(|&b| b == b' ')?;
455    let (patch_type, rest) = format_line.split_at(space);
456    let size_str = core::str::from_utf8(&rest[1..]).ok()?;
457    let size: u64 = size_str.parse().ok()?;
458
459    let kind = match patch_type {
460        b"literal" => BinaryBlockKind::Literal,
461        b"delta" => BinaryBlockKind::Delta,
462        _ => return None,
463    };
464
465    // Consume Base85 lines until blank line, tracking the end of actual data.
466    let data_start = parser.offset;
467    let mut data_end = data_start;
468    while let Some(line) = parser.next_line() {
469        if line.is_empty() {
470            break;
471        }
472        data_end = parser.offset;
473    }
474
475    // Slice the data lines, stripping the final line ending.
476    let data = &parser.input[data_start..data_end];
477    let data = data
478        .strip_suffix(b"\r\n".as_slice())
479        .or_else(|| data.strip_suffix(b"\n".as_slice()))
480        .unwrap_or(data);
481
482    Some(BinaryBlock {
483        kind,
484        data: BinaryData { size, data },
485    })
486}

488/// Decodes multi-line Base85 data with length indicators.
489///
490/// Each line has the format: `<len_c><data>`
491///
492/// From [5.1.1. Binary Payloads](https://diffx.org/spec/binary-diffs.html#binary-payloads):
493///
494/// > Each line represents up to 52 bytes of pre-encoded data.
495/// > There may be an unlimited number of lines.
496/// > They contain the following fields:
497/// >
498/// >  * `len_c` is a line length character.
499/// >     This encodes the length of the (pre-encoded) data written on this line.
500/// >  * `data` is Base85-encoded data for this line.
501#[cfg(feature = "binary")]
502fn decode_base85_lines(data: &[u8]) -> Result<Vec<u8>, BinaryPatchParseError> {
503    // A rough estimate: In Base85, 5 chars -> 4 bytes
504    let mut result = Vec::with_capacity(data.len() * 4 / 5);
505
506    for line in data.split(|&b| b == b'\n') {
507        let line = line.strip_suffix(b"\r".as_slice()).unwrap_or(line);
508        if line.is_empty() {
509            continue;
510        }
511
512        let length = decode_line_length(line[0])
513            .ok_or(BinaryPatchParseErrorKind::InvalidLineLengthIndicator)?;
514        let encoded = &line[1..];
515        let start = result.len();
516        base85::decode_into(encoded, &mut result)?;
517        result.truncate(start + length);
518    }
519
520    Ok(result)
521}

523/// Decodes a line length character to its numeric value.
524///
525/// From [Line Length Characters](https://diffx.org/spec/binary-diffs.html#line-length-characters):
526///
527/// > Each encoded line in a binary diff payload is prefixed by a line length character.
528/// > This encodes the length of the compressed (but not encoded) data for the line.
529/// >
530/// > Line length characters always represent a value between 1 and 52:
531/// >
532/// > * A value of A-Z represents a number between 1..26.
533/// > * A value of a-z represents a number between 27..52.
534#[cfg(feature = "binary")]
535fn decode_line_length(c: u8) -> Option<usize> {
536    match c {
537        b'A'..=b'Z' => Some((c - b'A' + 1) as usize),
538        b'a'..=b'z' => Some((c - b'a' + 27) as usize),
539        _ => None,
540    }
541}
